Syntax% [sF,v] = max(sF1,sF2)

Example

X = [2 8 4; 7 3 9]
max(X,[],1)
max(X,[],2)
max(X,5)

X =
     2     8     4
     7     3     9
ans =
     7     8     9
ans =
     8
     9
ans =
     5     8     5
     7     5     9
