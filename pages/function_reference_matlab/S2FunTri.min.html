Syntax% [sF,v] = min(sF1, sF2)

Example

X = [2 8 4; 7 3 9]
min(X,[],1)
min(X,[],2)
min(X,5)

X =
     2     8     4
     7     3     9
ans =
     2     3     4
ans =
     2
     3
ans =
     2     5     4
     5     3     5
