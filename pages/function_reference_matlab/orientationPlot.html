ODFSECTIONS

Syntax

oP = orientationPlot(ax,CS1,'param1',val1,'param2',val2)

Example

cs = crystalSymmetry('mmm')
oS = axisAngleSections(cs,cs)
ori = oS.makeGrid('resolution');
oM = PatalaColorKey(cs,cs)
rgb = oM.orientation2color(ori);
plot(oS,rgb,'surf')

 
cs = crystalSymmetry  
 
  symmetry: mmm    
  a, b, c : 1, 1, 1
 
oS = 
  axisAngleSections with properties:

           angles: [1×12 double]
      axesSectors: {1×12 cell}
          jointCS: [4×1 crystalSymmetry]
               oR: [1×1 orientationRegion]
    volumeScaling: 1
              CS1: [4×1 crystalSymmetry]
              CS2: [4×1 crystalSymmetry]
              tol: 0.0873
         plotGrid: []
         gridSize: []
               CS: [4×1 crystalSymmetry]
               SS: [4×1 crystalSymmetry]
oM = 
  PatalaColorKey with properties:

          CS1: [4×2 crystalSymmetry]
          CS2: [4×2 crystalSymmetry]
    antipodal: 1
