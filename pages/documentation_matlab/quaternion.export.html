Syntax

export(ori,fname)
export(ori,fname,'radians','Bunge')

fit ellipses to grains and store halfaxes and orientation in a struct
[S.angle,S.a,S.b] = fitEllipse(grains);

% store area
S.area = grains.area;

% export orientation and custom data
export(grains.meanOrientation,S)
